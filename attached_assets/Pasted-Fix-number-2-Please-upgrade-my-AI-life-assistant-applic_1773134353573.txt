function detectComplexity(message) {
  const text = message.toLowerCase().trim();
  const wordCount = text.split(/\s+/).length;
  let score = 0;

  if (wordCount < 12) return "LOW";

  const strongSignals = [
    "i don't know what",
    "should i quit",
    "important choice",
    "no hope",
    "losing myself",
    "don't see the point",
    "feels pointless",
    "not sure who i am"
  ];

  const mediumSignals = [
    "part of me",
    "can't seem to",
    "i keep",
    "been thinking about",
    "feeling lost",
    "feeling confused",
    "fear of",
    "pressure to",
    "getting worse",
    "keeps coming back",
    "career decision",
    "wrong direction",
    "what if i",
    "my future",
    "stuck in"
  ];

  for (const s of strongSignals) {
    if (text.includes(s)) {
      score += 2;
      if (score >= 3) return "HIGH";
    }
  }

  for (const s of mediumSignals) {
    if (text.includes(s)) {
      score += 1;
      if (score >= 3) return "HIGH";
    }
  }

  if (wordCount > 80) score += 2;
  if (wordCount > 140) score += 1;

  return score >= 3 ? "HIGH" : "LOW";
}

function buildMemoryContext(memory) {

    return "";
