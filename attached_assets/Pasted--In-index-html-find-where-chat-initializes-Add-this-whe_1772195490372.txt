    const response = await fetch('/api/opening-message', {

      credentials: 'include'
    });
    
    const data = await response.json();
    
    if (data.message) {
      // Display as first AI message in chat

      displayAIMessage(data.message.text);
      
      // If message has chips, show them

        showPromptChips(data.message.chips);
      }
    }

}
