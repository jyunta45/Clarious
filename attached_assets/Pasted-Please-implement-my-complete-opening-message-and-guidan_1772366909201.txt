// ── PUBLIC WRAPPER (safe entry point) ─────────
// This is the ONLY function called from outside.
// Wraps everything in try/catch.
// Handles all failures silently.
// Prevents repeat messages.

export function buildOpeningMessage(params) {
  try {
    const opening = _buildOpeningMessage(params);

    // GUARD 1 — Do not repeat yesterday's message
    if (opening.text === params.userData?.lastOpeningMessage) {
      return buildFallbackOpening(
        params.userData?.lastOpeningMessage
      );
    }

    return opening;

  } catch (error) {
    // GUARD 2 — Silent fail, never show blank screen
    console.error("Opening message build failed:", error);
    return buildFallbackOpening(
      params.userData?.lastOpeningMessage
    );
  }
}

// ── PRIVATE BUILDER ───────────────────────────
// Priority order (top wins):
// 1. Returning user (2+ days away)
// 2. Guidance days 1-7 (any time of day)
// 3. Graduation day 8 (fires once, any time)
// 4. Missed habit (post-graduation only)
// 5. Time-based defaults

function _buildOpeningMessage({
  userMemory,
  sessionSummary,
  habits,
  lastActiveAt,
  guidanceDay,
  userData
}) {
  const now      = new Date();
  const hour     = now.getHours();
  const daysAway = lastActiveAt
    ? (Date.now() - new Date(lastActiveAt)) / 86400000
    : 0;
  const isMorning = hour >= 5 && hour <= 11;

  // ── PRIORITY 1 — RETURNING USER ──────────────
  // Life context always wins over everything.
  // If away 2+ days, acknowledge it first.

  if (daysAway >= 2) {
    const days = Math.floor(daysAway);
    if (sessionSummary?.focus) {
      return {
        text: `You've been away for ${days} day${days > 1 ? 's' : ''}.\n\nLast time you were working on ${sessionSummary.focus}.\n\nWhere does that stand now?`,
        type: "returning",
        chips: [
          "Catch you up on what happened",
          "I need to get back on track",
          "Something changed",
          "Let's review my direction"
        ]
      };
    }
    return {
      text: `You've been away for ${days} day${days > 1 ? 's' : ''}.\n\nWhat's been taking your attention?`,
      type: "returning",
      chips: [
        "A lot happened",
        "I need to reset",
        "Something changed for me",
        "Just checking back in"
      ]
    };
  }

  // ── PRIORITY 2 — GUIDANCE DAYS 1-7 ───────────
  // Fires any time of day — no morning restriction.
  // Habit check intentionally skipped during this phase.
  // Teaching is more important than accountability
  // during the first 7 active engaged days.

  if (guidanceDay >= 1 && guidanceDay <= 7) {
    const message = guidanceMessages[guidanceDay];
    return {
      text:  message.text,
      type:  "guidance",
      day:   guidanceDay,
      chips: message.chips
    };
  }

  // ── PRIORITY 3 — GRADUATION DAY 8 ────────────
  // Fires exactly once at any time of day.
  // After firing, endpoint sets guidanceDay to 9
  // so this block never executes again.

  if (guidanceDay === 😎 {
    return {
      text:  guidanceMessages[8].text,
      type:  "transition",
      chips: guidanceMessages[8].chips
    };
  }

  // ── PRIORITY 4 — MISSED HABIT ─────────────────
  // Only runs post-graduation (guidanceDay >= 9).
  // Only fires for established users with real streaks.
  // Protects new users and guidance-phase users.

  const userEstablished = userMemory?.goals?.length > 0;
  const missedHabit = userEstablished &&
    habits?.find(h =>
      h.completedYesterday === false &&
      h.streak > 2
    );

  if (missedHabit) {
    return {
      text: `You missed "${missedHabit.name}" yesterday.\n\nNot a judgment — what got in the way?`,
      type: "habit",
      chips: [
        "Reset momentum today",
        "Let's figure out what happened",
        "Adjust the habit",
        "Something else came up"
      ]
    };
  }

  // ── PRIORITY 5 — TIME-BASED DEFAULTS ─────────
  // Memory-aware where possible.
  // Falls back to generic if no memory yet.

  if (hour >= 0 && hour < 5) {
    return {
      text: `You're up late.\n\nWhat's on your mind?`,
      type: "latenight",
      chips: [
        "Can't sleep — something's on my mind",
        "Working late on something",
        "Just needed to talk",
        "Help me wind down"
      ]
    };
  }

  if (isMorning) {
    const focus = userMemory?.goals?.slice(-1)[0] || null;
    return {
      text: focus
        ? `Morning. You're working toward ${focus}.\n\nWhat's today's move?`
        : `What matters most to move forward today?`,
      type: "morning",
      chips: [
        "Help me plan today",
        "I'm struggling with something",
        "Accountability check",
        "Just talk"
      ]
    };
  }

  if (hour >= 12 && hour <= 17) {
    return {
      text: `How is today tracking compared to what you intended?`,
      type: "afternoon",
      chips: [
        "I'm behind on something",
        "Need to refocus",
        "Decision I'm sitting on",
        "Check in on my habits"
      ]
    };
  }

  return {
    text: `What's one thing that moved forward today —\nand one thing worth carrying into tomorrow?`,
    type: "evening",
    chips: [
      "Review my day",
      "Something's bothering me",
      "Tomorrow's intention",
      "What patterns do you see?"
    ]
  };
}

Find updateGuidanceDay() in continuityEngine.js or server.js.

async function updateGuidanceDay(userData) {

  // Already graduated — nothing to do
  if (!userData.guidanceMode) return userData;

  const today = new Date().toDateString();

  // Already counted today — no double increment
  if (userData.lastGuidanceDate === today) return userData;

  // GUARD — Pause after 2+ day absence
  // Guidance reflects real engagement not calendar days
  const daysAway = userData.lastActiveAt
    ? (Date.now() - new Date(userData.lastActiveAt)) / 86400000
    : 0;

  if (daysAway >= 2) {
    userData.lastActiveAt = Date.now();
    return userData;  // Pause — do not advance
  }

  // GUARD — Freeze if user opened app but never talked
  // Sending a message sets userSentMessageToday = true
  // Opening the app alone does not count as engagement
  if (!userData.userSentMessageToday) {
    return userData;  // Freeze — do not advance
  }

  // All guards passed — advance guidance day
  userData.guidanceDay          = (userData.guidanceDay || 0) + 1;
  userData.lastGuidanceDate     = today;
  userData.lastActiveAt         = Date.now();
  userData.userSentMessageToday = false;

  // Graduation — user completed 7 active engaged days
  if (userData.guidanceDay > 7) {
    userData.guidanceMode = false;
    userData.guidanceDay  = 8;

    // Silent graduation memory write
    // Unlocks collaborative_autonomous leadership style
    // AI becomes more direct and peer-level after this
    if (!userData.userMemory)
      userData.userMemory = {};
    if (!userData.userMemory.meta)
      userData.userMemory.meta = {};

    userData.userMemory.meta.guidanceCompleted   = true;
    userData.userMemory.meta.guidanceCompletedAt = new Date().toISOString();
    userData.userMemory.meta.leadershipStyle     = "collaborative_autonomous";

    await db.update(userDataTable)
      .set({
        guidanceMode: false,
        guidanceDay:  8,
        userMemory:   userData.userMemory
      })
      .where(eq(userDataTable.userId, userData.userId));
  }

  return userData;
}
