// ======================================
// CONTEXT COMPRESSION ENGINE
// ======================================

/*
Goal:
Replace expensive raw history
with summary + recent messages
*/

function buildContext({
  enhancedSystem,
  conversation,
  rollingSummary
}) {

  const TURN_THRESHOLD = 6;
  const RECENT_LIMIT = 4;

  let messages = [];

  // Always include system prompt
  messages.push({
    role: "system",
    content: enhancedSystem
  });

  // ----------------------------------
  // EARLY CONVERSATION
  // send normal history
  // ----------------------------------
  if (conversation.length <= TURN_THRESHOLD) {

    messages.push(...conversation);

    return messages;
  }

  // ----------------------------------
  // COMPRESSED MODE
  // ----------------------------------

  // Inject rolling summary
  if (
    rollingSummary &&
    rollingSummary !==
      "No significant session context to carry forward."
  ) {
    messages.push({
      role: "system",
      content:
        "Conversation summary:\n" + rollingSummary
    });
  }

  // Send only recent turns
  const recentMessages =
    conversation.slice(-RECENT_LIMIT);

  messages.push(...recentMessages);

  return messages;
}

module.exports = {
  buildContext
};
